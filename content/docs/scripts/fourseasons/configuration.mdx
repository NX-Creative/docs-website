All settings live in `shared/config.lua`. The file is in `escrow_ignore`
so it remains editable after install. Event name constants and locale
strings live in `shared/constants.lua` and the `Config.Locale` table
respectively.

## Gameplay flooding

Subtle low-street flooding tied to the weather cycle. Separate from
the admin-only extreme flooding used for testing. Requires
`Config.StormSystems.Enabled = true`.

<ConfigOption name="Config.GameplayFlooding.Enabled" type="boolean" default="false" description="Master toggle for gameplay flooding." />

<ConfigOption name="Config.GameplayFlooding.AutoReset" type="boolean" default="true" description="Automatically reset water after the profile duration expires." />

<ConfigOption name="Config.GameplayFlooding.CooldownMinutes" type="number" default="180" description="Minimum time between gameplay flood triggers." />

<ConfigOption name="Config.GameplayFlooding.DurationCheckIntervalSeconds" type="number" default="30" description="How often the server checks whether the active flood should expire." />

<ConfigOption name="Config.GameplayFlooding.Profiles" type="table" description="Flood profile definitions keyed by id. Shipped: subtle, moderate, strong. Each profile: { maxFloodHeight, increaseRate, threadWait, durationMinutes }." />

<ConfigOption name="Config.GameplayFlooding.ProfileWeights" type="table" default="{ subtle = 0.78, moderate = 0.20, strong = 0.02 }" description="Weighted selection when a profile is picked automatically." />

<ConfigOption name="Config.GameplayFlooding.SeasonRules" type="table" description="Per-season trigger rules. Shipped: SPRING (THUNDER, minIntensity 0.72, triggerChance 0.35) and FALL (RAIN or THUNDER, minIntensity 0.80, triggerChance 0.20)." />

### Blackout

<ConfigOption name="Config.StormSystems.Blackout.Enabled" type="boolean" default="true" />

<ConfigOption name="Config.StormSystems.Blackout.MinIntensity" type="number" default="0.72" description="Fallback intensity threshold for weather types not listed in EligibleWeather." />

<ConfigOption name="Config.StormSystems.Blackout.EligibleWeather" type="table" default="{ THUNDER = { minIntensity = 0.72 }, RAIN = { minIntensity = 0.88 }, BLIZZARD = { minIntensity = 0.72 } }" description="Per-weather minimum intensity required for blackout pulses to fire." />

<ConfigOption name="Config.StormSystems.Blackout.PulseMinSeconds" type="number" default="6" />
<ConfigOption name="Config.StormSystems.Blackout.PulseMaxSeconds" type="number" default="14" />

<ConfigOption name="Config.StormSystems.Blackout.DurationTiers" type="table" default="{}" description="Optional weighted duration tiers per weather type. When a weather type has tiers, each blackout pulse picks one at random (weighted) instead of using PulseMin/MaxSeconds. Each tier is { duration = seconds, weight = N } for a fixed value, or { min = seconds, max = seconds, weight = N } for a random range. Durations are clamped to [2, 300] seconds. Leave a weather key absent to keep the default short-pulse behavior. Example for dramatic thunderstorm outages: DurationTiers = { THUNDER = { { duration = 30, weight = 60 }, { duration = 60, weight = 30 }, { duration = 300, weight = 10 } } }." />

<ConfigOption name="Config.StormSystems.Blackout.CooldownMinSeconds" type="number" default="20" />
<ConfigOption name="Config.StormSystems.Blackout.CooldownMaxSeconds" type="number" default="45" />

<ConfigOption name="Config.StormSystems.Blackout.VehicleLights" type="boolean" default="false" description="When false, vehicle lights are disabled during blackout pulses." />
